Cache Simulator


          void run_cpu(CPU* cpu) {
            TraceLine trace_line;
            while (read_address(cpu, &trace_line) != EOF) {
              cpu->address_count++;
              AccessResult result = cache_access(cpu->cache, &trace_line);
              if (result == HIT) {
                cpu->hits++;
              }
              else if (result == COLD_MISS) {
                cpu->cold++;
              }   
              else {
                cpu->conflict++;
              }
            }
          }
